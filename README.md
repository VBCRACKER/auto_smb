# auto_smb
a python script using shodan cli to search for authentication disabled SMB and enumerate shares.

# Installation
apt install python3-shodan,
apt install smbclient,
pip install -r requirements.txt,
shodan init YOUR_API_KEY

# Run the script
python3 auto_smb.py
